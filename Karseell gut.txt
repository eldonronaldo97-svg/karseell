<script>
// اختيار العرض
let selectedOffer = null;
document.querySelectorAll('.offer').forEach(offer => {
  offer.addEventListener('click', function() {
    document.querySelectorAll('.offer').forEach(o => o.classList.remove('selected'));
    this.classList.add('selected');
    selectedOffer = this;
  });
});

// إرسال الطلب
document.getElementById("orderForm").addEventListener("submit", function(e) {
  e.preventDefault();

  if (!selectedOffer) {
    alert("من فضلك اختر أحد العروض أولاً!");
    return;
  }

  const name = document.getElementById("name").value.trim();
  const phone = document.getElementById("phone").value.trim();
  const city = document.getElementById("city").value.trim();
  const address = document.getElementById("address").value.trim();

  if (!name || !phone || !city || !address) {
    alert("من فضلك أكمل كل البيانات قبل الإرسال ✅");
    return;
  }

  // تتبع الحدث في فيسبوك مع قيمة العرض
  fbq('track', 'Purchase', {
    value: parseFloat(selectedOffer.dataset.price),
    currency: 'EGP'
  });

  // إظهار رسالة النجاح
  document.getElementById("successMessage").style.display = "block";

  // إعادة ضبط الحقول
  e.target.reset();
  selectedOffer.classList.remove('selected');
  selectedOffer = null;
});
</script>
